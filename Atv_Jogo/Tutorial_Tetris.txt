Passo 3: Elevando o Estado (Lógica de Colisão)Para coletar dados do tabuleiro e da peça atual simultaneamente, o estado principal fica centralizado em um único objeto controlador (player). A função collide lê a posição do jogador em tempo real e calcula o impacto contra as bordas e outras peças fixas.

-----------------------Script-----------------------
const player = {
    pos: {x: 0, y: 0},
    matrix: null,
    score: 0,
};

function collide(arena, player) {
    const [m, o] = [player.matrix, player.pos];
    for (let y = 0; y < m.length; ++y) {
        for (let x = 0; x < m[y].length; ++x) {
            if (m[y][x] !== 0) {
                // Bloqueia saídas pelas laterais esquerda e direita
                if (x + o.x < 0 || x + o.x >= COLS) {
                    return true;
                }
                // Ignora colisões acima do teto visível, mas valida o chão do tabuleiro
                if (y + o.y < 0) {
                    continue;
                }
                if (!arena[y + o.y] || arena[y + o.y][x + o.x] !== 0) {
                    return true;
                }
            }
        }
    }
    return false;
}

function merge(arena, player) {

function playerReset() {
    const pieces = 'ILJOSZMT';
    player.matrix = createPiece(pieces[pieces.length * Math.random() | 0]);
    player.pos.y = 0;
    player.pos.x = (arena[0].length / 2 | 0) - (player.matrix[0].length / 2 | 0);
    
    // Game Over
    if (collide(arena, player)) {
        arena.forEach(row => row.fill(0));

        updateScore();

// Gira a peça (Matriz)
function playerRotate(dir) {
    const pos = player.pos.x;
    let offset = 1;
    rotate(player.matrix, dir);
    while (collide(arena, player)) {
        player.pos.x += offset;
        offset = -(offset + (offset > 0 ? 1 : -1));
        if (offset > player.matrix[0].length) {
            rotate(player.matrix, -dir);
            player.pos.x = pos;
            return;
        }
    }
}

// Lógica matemática para rotacionar matriz
function rotate(matrix, dir) {
    for (let y = 0; y < matrix.length; ++y) {
        for (let x = 0; x < y; ++x) {
            [
                matrix[x][y],
                matrix[y][x],
            ] = [
                matrix[y][x],
                matrix[x][y],
            ];
        }
    }
    if (dir > 0) {
        matrix.forEach(row => row.reverse());
    } else {
        matrix.reverse();
    }
}

function collide(arena, player) {
    const [m, o] = [player.matrix, player.pos];
    for (let y = 0; y < m.length; ++y) {
        for (let x = 0; x < m[y].length; ++x) {
            if (m[y][x] !== 0) {
                // Bloqueia saídas pelas laterais esquerda e direita
                if (x + o.x < 0 || x + o.x >= COLS) {
                    return true;
                }
                // Ignora colisões acima do teto visível, mas valida o chão do tabuleiro
                if (y + o.y < 0) {
                    continue;
                }
                if (!arena[y + o.y] || arena[y + o.y][x + o.x] !== 0) {
                    return true;
                }
            }
        }
    }
    return false;
}

Passo 5: Interatividade e Time Travel (Loop e Reinício)O loop de animação (requestAnimationFrame) atualiza constantemente a passagem de tempo delta (deltaTime). Quando ocorre um Game Over real, o estado do tempo é reiniciado por completo usando performance.now(), limpando o tabuleiro e permitindo uma nova partida sem retenção de lixo de memória anterior.

let dropCounter = 0;

let lastTime = 0;
let gameOver = false;

function update(time = 0) {
    const deltaTime = time - lastTime;
    lastTime = time;

    if (!gameOver) {
        dropCounter += deltaTime;
        if (dropCounter > dropInterval) {
            playerDrop(); // Move a peça para baixo
        }
    }

function resetGame() {
    arena.forEach(row => row.fill(0)); // Reseta a matriz lógica
    player.score = 0;
    gameOver = false;
    document.getElementById('game-over-screen').style.display = 'none';
    updateScore();
    playerReset();
    lastTime = performance.now(); // Zera o relógio para evitar pulos indesejados de peças
    dropCounter = 0;
